class HomeServe(TemplateView):
            template_name = 'home.html'  #page path in tamplates folder
    
            #context
            def get_context_data(self, **kwargs):
                context = super().get_context_data(**kwargs)
                context['title'] ="סייבר"  # add context title with the = var to the serve / in html you will call {{title}} to get the var/array/object|s
                return context
